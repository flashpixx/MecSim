102  /**
103  * stores the benchmark data into a Json file
104  */
105  public void store()
106  {
107  if ( ( m_filename == null ) || ( m_filename.isEmpty() ) )
108  return;
109 
110  final Map<String, Map<String, Double>> l_benchmark = m_result.entrySet().parallelStream()
111  .map( i ->
112  new AbstractMap.SimpleImmutableEntry<>(
113  i.getKey(),
114  Collections.unmodifiableMap( new HashMap<String, Double>()
115  {{
116  put( "max", i.getValue().getMax() );
117  put( "min", i.getValue().getMin() );
118  put( "kurtosis", i.getValue().getKurtosis() );
119  put( "arithmetic mean", i.getValue().getMean() );
120  put( "geometric mean", i.getValue().getGeometricMean() );
121  put( "50-percentile", i.getValue().getPercentile( 50 ) );
122  put( "25-percentile", i.getValue().getPercentile( 25 ) );
123  put( "75-percentile", i.getValue().getPercentile( 75 ) );
124  put( "standard deviation", i.getValue().getStandardDeviation() );
125  put( "skewness", i.getValue().getSkewness() );
126  put( "count", new Double( i.getValue().getN() ) );
127  put( "sum", i.getValue().getSum() );
128  put( "sum square", i.getValue().getSumsq() );
129  put( "variance", i.getValue().getVariance() );
130  }} )
131  )
132  )
133  .collect( Collectors.toMap( Map.Entry::getKey, Map.Entry::getValue ) );
134 
135  if ( l_benchmark.isEmpty() )
136  return;
137 
138  try
139  {
140  FileUtils.writeStringToFile( new File( m_filename ), CCommon.toJson( l_benchmark ), "UTF-8" );
141  }
142  catch ( final IOException l_exception )
143  {
144  CLogger.error( l_exception );
145  }
146  }
147 
148 }
