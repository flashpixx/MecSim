81  @Override
82  public Set<ICar> generate( final Collection<Pair<GeoPosition, GeoPosition>> p_waypoints, final int p_count )
83  {
84  final ArrayList<Pair<EdgeIteratorState, Integer>> l_cells = this.generateRouteCells( p_waypoints );
85  return IntStream.range( 0, p_count )
86  .parallel()
87  .mapToObj( i -> this.getCar( l_cells ) )
88  .collect( Collectors.toCollection( () -> new HashSet<ICar>() ) );
89  }

91  /**
92  * creates the route cells
93  *
94  * @param p_waypoints waypoint pair list
95  * @return cell list
96  *
97  * @bug run in parallel with stream
98  */
99  protected final ArrayList<Pair<EdgeIteratorState, Integer>> generateRouteCells( final Collection<Pair<GeoPosition, GeoPosition>> p_waypoints )
100  {
101  final ArrayList<Pair<EdgeIteratorState, Integer>> l_cells = new ArrayList<>();
102  for ( final Pair<GeoPosition, GeoPosition> l_point : p_waypoints )
103  {
104  final List<List<EdgeIteratorState>> l_route = m_graph.getRoutes( l_point.getLeft(), l_point.getRight(), m_weight, 1 );
105  if ( l_route.size() > 0 )
106  l_cells.addAll( m_graph.getRouteCells( l_route.get( 0 ) ) );
107  }
108 
109  return l_cells;
110  }
