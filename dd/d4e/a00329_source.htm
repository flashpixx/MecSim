50  // data must be added in the method, because the instantiation of the map is called before the child object is instantiate,

52  if ( !m_inspect.containsKey( CCommon.getResourceString( IInspectorDefault.class, "objectid" ) ) )
53  m_inspect.put( CCommon.getResourceString( IInspectorDefault.class, "objectid" ), this.hashCode() );
54 
55  if ( !m_inspect.containsKey( CCommon.getResourceString( IInspectorDefault.class, "classname" ) ) )
56  m_inspect.put( CCommon.getResourceString( IInspectorDefault.class, "classname" ), CCommon.removePackageName( this.getClass().getName() ) );
57
