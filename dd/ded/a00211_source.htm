185  * @param p_data input data
186  */
187  private final void web_static_write( final Map<String, Object> p_data ) throws IOException
188  {
189  if ( CSimulation.getInstance().isRunning() )
190  throw new IllegalStateException( CCommon.getResourceString( this, "running" ) );
191  if ( !p_data.containsKey( "source" ) )
192  throw new IllegalArgumentException( CCommon.getResourceString( this, "noagentdata" ) );
193 
194  switch ( m_type )
195  {
196  case Jason:
197  FileUtils.writeStringToFile( IEnvironment.getAgentFile( this.getAgentName( p_data ) ), (String) p_data.get( "source" ) );
198  break;
199 
200  default:
201  }
202  }
