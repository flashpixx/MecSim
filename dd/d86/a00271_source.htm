64  @Override
65  public void mouseClicked( final MouseEvent p_event )
66  {
67  if ( ( CSimulation.getInstance().isRunning() ) || ( !SwingUtilities.isLeftMouseButton( p_event ) ) || ( p_event.getClickCount() != 2 ) )
68  return;
69 
70 
71  final COSMViewer l_viewer = (COSMViewer) p_event.getSource();
72 
73  try
74  {
75  switch ( l_viewer.getCurrentClickableLayer() )
76  {
77  case Source:
78  final CCarWayPointLayer l_layer = CSimulation.getInstance().getWorld().<CCarWayPointLayer>getTyped( "Car WayPoints" );
79  boolean l_isfound = false;
80 
81  for ( final IWayPoint<ICar> l_item : l_layer )
82  if ( isinBall( p_event.getPoint(), l_viewer.convertGeoPositionToPoint( l_item.getPosition() ) ) )
83  {
84  l_isfound = true;
85  l_layer.remove( l_item );
86  break;
87  }
88 
89  if ( !l_isfound )
90  CSimulation.getInstance().getStorage().<CWaypointEnvironment>get( "waypoint" ).setWaypoint(
91  l_viewer.getViewpointGeoPosition(
92  p_event.getPoint()
93  )
94  );
95  break;
96 
97 
98  case ForbiddenEdge:
99  // read graph & weight data on-fly, because on loading simulation data the graph instance can be changed
100  CSimulation.getInstance().getWorld().<CCarLayer>getTyped( "Cars" ).getGraph().<CForbiddenEdge>getWeight(
101  CGraphHopper.EWeight.ForbiddenEdge
102  ).swap();
103 
104  l_viewer.repaint();
105  break;
106  }
107  }
108  catch ( final Exception l_exception )
109  {
110  CLogger.out( l_exception );
111  }
112 
113  }

115  @Override
116  public void mouseMoved( final MouseEvent p_event )
117  {
118  final COSMViewer l_viewer = (COSMViewer) p_event.getSource();
119 
120  // read graph & weight data on-fly, because on loading simulation data the graph instance can be changed
121  final CGraphHopper l_graph = CSimulation.getInstance().getWorld().<CCarLayer>getTyped( "Cars" ).getGraph();
122  final CForbiddenEdge l_weight = l_graph.<CForbiddenEdge>getWeight( CGraphHopper.EWeight.ForbiddenEdge );
123 
124  if ( !l_viewer.getCurrentClickableLayer().equals( COSMViewer.EClickableLayer.ForbiddenEdge ) )
125  l_weight.clearReserve();
126  else
127  l_weight.reserve(
128  l_graph.getClosestEdge(
129  l_viewer.getViewpointGeoPosition(
130  p_event.getPoint()
131  )
132  )
133  );
134 
135  l_viewer.repaint();
136  }

138  /**
139  * checks if a point is in a radius around another
140  *
141  * @param p_click point of the click
142  * @param p_center point of the center
143  * @return bool if the click is within the radius
144  */
145  private static boolean isinBall( final Point p_click, final Point2D p_center )
146  {
147  final double p_xval = p_click.getX() - p_center.getX();
148  final double p_yval = p_click.getY() - p_center.getY();
149 
150  return Math.sqrt( p_xval * p_xval + p_yval * p_yval ) <= s_clickradius;
151  }
