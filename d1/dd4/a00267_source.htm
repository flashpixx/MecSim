59  /**
60  * checks the acceleration and increment the speed
61  *
62  * @param p_layer car layer
63  * @param p_car car object
64  */
65  protected final void checkAccelerationWithEdgeSpeed( final CCarLayer p_layer, final ICar p_car )
66  {
67  p_car.setCurrentSpeed(
68  Math.min(
69  Math.min( p_car.getMaximumSpeed(), (int) p_layer.getGraph().getEdgeSpeed( p_car.getEdge() ) ),
70  p_car.getCurrentSpeed() + (int) p_layer.getUnitConvert().getAccelerationToSpeed( p_car.getAcceleration() )
71  )

73  }

75  /**
76  * checks of a collision and reduce speed
77  *
78  * @param p_layer car layer
79  * @param p_car car object
80  */
81  protected void checkCollision( final CCarLayer p_layer, final ICar p_car )
82  {
83  final Map<Double, ICar> l_predecessor = p_car.getPredecessor();
84  if ( ( l_predecessor != null ) && ( l_predecessor.size() > 0 ) )
85  {
86  // get distance which can be drive in one step and distance to the predecessor
87  final double l_speeddistance = p_layer.getUnitConvert().getSpeedToDistance( p_car.getCurrentSpeed() );
88  final double l_distance = l_predecessor.entrySet().iterator().next().getKey();
89 
90  if ( l_distance < l_speeddistance )
91  p_car.setCurrentSpeed( Math.max( 0, p_layer.getUnitConvert().getSpeedOfDistance( l_speeddistance - l_distance ) ) );
92  }
93  }
94 
95  /**
96  * checks the linger probability and modify speed
97  *
98  * @param p_layer car layer
99  * @param p_car car object
100  */
101  protected final void checkLinger( final CCarLayer p_layer, final ICar p_car )
102  {
103  if ( ( p_car.getCurrentSpeed() > 0 ) && ( m_random.nextDouble() <= p_car.getLingerProbability() ) )
104  p_car.setCurrentSpeed(
105  Math.max(
106  c_minimalspeed, p_car.getCurrentSpeed() - (int) p_layer.getUnitConvert().getAccelerationToSpeed(
107  p_car.getDeceleration()
108  )
109  )
110  );
111  }
