94  /**
95  * navigates one item back
96  */
97  public final void back()
98  {
99  if ( m_webview.getEngine().getHistory().getCurrentIndex() < 1 )
100  return;
101 
102  m_webview.getEngine().getHistory().go( -1 );
103  }
104 
105  /**
106  * navigates on item forward
107  */
108  public final void forward()
109  {
110  if ( m_webview.getEngine().getHistory().getCurrentIndex() >= m_webview.getEngine().getHistory().getEntries().size() - 1 )
111  return;
112 
113  m_webview.getEngine().getHistory().go( +1 );
114  }

192  /**
193  * sets the mennubar
194  *
195  * @param p_menu menu settings
196  * @param p_menu menu options
197  * @return column number
198  */
199  private int setMenuBar( final EMenu p_menu )
200  {
201  this.setVgap( 5 );
202  this.setHgap( 5 );
203 
204  final int l_return;
205  switch ( p_menu )
206  {
207  case BackForward:
208  l_return = this.setBackForward( 0 );
209  break;
210 
211  case URL:
212  l_return = this.setURLMenu( 0 );
213  break;
214 
215  case Full:
216  l_return = this.setBackForward( this.setURLMenu( 0 ) );
217  break;
218 
219  default:
220  l_return = 1;
221  }
222 
223  return l_return;
224  }

251  final EventHandler<ActionEvent> l_buttonaction = new EventHandler<ActionEvent>()

267 
268  this.getChildren().addAll( l_url, l_button );
