306  *
307  * @param p_data input data
308  */
309  private final void web_static_set( final Map<String, Object> p_data )
310  {
311  // create travable map and clear cached settings
312  m_currentsettings.clear();
313  final CNameHashMap.CImmutable l_data = new CNameHashMap.CImmutable( p_data );
314 
315  // create items from the UI data and set internal properties
316  m_currentsettings.put( "name", l_data.<String>get( "name" ) );
317  m_currentsettings.put( "waypoint", l_data.<String>get( "waypoint" ) );
318  m_currentsettings.put( "radius", l_data.<Number>get( "radius" ).doubleValue() );
319  m_currentsettings.put(
320  "color", new Color(
321  l_data.<Number>get( "color/red" ).intValue(),
322  l_data.<Number>get( "color/green" ).intValue(),
323  l_data.<Number>get( "color/blue" ).intValue()
324  )
325  );
326 
327  m_currentsettings.put(
328  "factory",
329  EFactory.valueOf( l_data.<String>get( "factory" ) ).get(
330  l_data.<Number>get( "speedfactor" ).doubleValue(),
331  this.createDistribution( l_data.get( "distribution/maxspeed" ) ),
332  this.createDistribution( l_data.get( "distribution/acceleration" ) ),
333  this.createDistribution( l_data.get( "distribution/deceleration" ) ),
334  this.createDistribution( l_data.get( "distribution/linger" ) ),
335  CGraphHopper.EWeight.valueOf( l_data.<String>get( "weight" ) ),
336  l_data.<String>get( "agent/agent" ),
337  EAgentLanguages.valueOf( l_data.<String>get( "agent/type" ) )
338  )
339  );
340 
341  m_currentsettings.put(
342  "generator",
343  EGenerator.valueOf( l_data.<String>getOrDefault( "generator", "" ) ).get(
344 
345  this.createDistribution( l_data.get( "distribution/generator" ) ),
346  l_data.<Number>get( "carcount" ).intValue()
347 
348  )
349  );
350

388  /**
389  * returns the factory with the current settings
390  *
391  * @param p_data array with all parameters (depend on factory call)
392  * @return factory
393  */
394  @SuppressWarnings( "unchecked" )
395  public final IFactory<?> get( final Object... p_data )
396  {
397  switch ( this )
398  {
399  case DefaultCarFactory:
400  return new CDefaultCarFactory(
401  // maxspeed factor
402  (Double) p_data[0],
403  // maxspeed distribution
404  (AbstractRealDistribution) p_data[1],
405  // acceleration distribution
406  (AbstractRealDistribution) p_data[2],
407  // deceleration distribution
408  (AbstractRealDistribution) p_data[3],
409  // linger distribution
410  (AbstractRealDistribution) p_data[4],
411  // routing weight
412  (CGraphHopper.EWeight) p_data[5]
413  );
414 
415  case DefaultAgentCarFactory:
416  return new CAgentCarFactory(
417  // maxspeed factor
418  (Double) p_data[0],
419  // maxspeed distribution
420  (AbstractRealDistribution) p_data[1],
421  // acceleration distribution
422  (AbstractRealDistribution) p_data[2],
423  // deceleration distribution
424  (AbstractRealDistribution) p_data[3],
425  // linger distribution
426  (AbstractRealDistribution) p_data[4],
427  // routing weight
428  (CGraphHopper.EWeight) p_data[5],
429  // agent name
430  (String) p_data[6],
431  // agent type
432  (EAgentLanguages) p_data[7]
433  );
434 
435  default:
436  throw new IllegalStateException( CCommon.getResourceString( EFactory.class, "unknownfactory" ) );
437  }
