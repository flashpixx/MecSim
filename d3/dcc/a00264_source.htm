75  /**
76  * pushs a message to the queue
77  *
78  * @param p_receiverpath receiver
79  * @param p_message message
80  */
81  public final synchronized void pushMessage( final CPath p_receiverpath, final IMessage<?> p_message )
82  {
83  if ( ( p_receiverpath == null ) || ( p_message == null ) || ( p_receiverpath.isEmpty() ) )
84  return;
85 
86  if ( !m_root.pathExist( p_receiverpath ) )
87  {
88  CLogger.error( CCommon.getResourceString( this, "messagefail", p_message.getData(), p_receiverpath ) );
89  return;
90  }
91 
92  if ( ( p_message.getSource() == null ) || ( p_message.getSource().isEmpty() ) || ( p_message.getSource().getPath().equals( p_receiverpath ) ) )
93  {
94  CLogger.error( CCommon.getResourceString( this, "push", p_message, p_receiverpath ) );
95  return;
96  }
97 
98  // check time to live value
99  if ( p_message.ttl() < 0 )
100  {
101  CLogger.info( CCommon.getResourceString( this, "ttl", p_message ) );
102  return;
103  }
104 
105  // receive message / push message to the receiver - increment main receiver of the message
106  this.incrementMessageFlow( p_message.getSource(), p_receiverpath );
107  for ( final Pair<Set<IParticipant>, Set<IMessage>> l_item : m_root.getNode( p_receiverpath ).getTreeData() )
108  {
109  // if item equal null skip
110  if ( l_item == null )
111  continue;
112 
113  l_item.getRight().add( p_message );
114 
115  // all subnodes of the receiver gets the message from the receiver the root node get the receiver from the message sender
116  for ( final IParticipant l_subreceiver : l_item.getLeft() )
117  this.incrementMessageFlow( p_receiverpath, l_subreceiver.getReceiverPath() );
118  }
119 
120  // call listener with receiver and message
121  for ( final IActionListener l_item : m_listener )
122  l_item.onPushMessage( p_receiverpath, p_message );
123  }

125  /**
126  * register a new participant
127  *
128  * @param p_path path of the receiver
129  * @param p_receiver participant
130  */
131  public final synchronized void register( final CPath p_path, final IParticipant p_receiver )
132  {
133  if ( ( p_path == null ) || ( p_path.isEmpty() ) || ( p_receiver == null ) || ( p_receiver == null ) )
134  {
135  CLogger.error( CCommon.getResourceString( this, "register", p_receiver, p_path ) );
136  return;
137  }
138 
139  final CTreeNode<Pair<Set<IParticipant>, Set<IMessage>>> l_node = m_root.getNode( p_path );
140  if ( l_node.isDataNull() )
141  l_node.setData( new ImmutablePair<>( new HashSet<>(), new HashSet<>() ) );
142  l_node.getData().getLeft().add( p_receiver );
143 
144  CLogger.info( CCommon.getResourceString( this, "registered", p_receiver, p_path ) );
145 
146  for ( final IActionListener l_item : m_listener )
147  l_item.onRegister( p_path, p_receiver );
148  }

166  @Override
167  public final void step( final int p_currentstep, final ILayer p_layer ) throws Exception
168  {
169  for ( final Pair<Set<IParticipant>, Set<IMessage>> l_item : m_root.getTreeData( false ) )
170  {
171  // data element within the tree can be used null values, so this items will be skipped
172  // the item is also skipped, if there does not exists messages
173  if ( ( l_item == null ) || ( l_item.getRight() == null ) || ( l_item.getRight().isEmpty() ) )
174  continue;
175 
176  for ( final IParticipant l_receiver : l_item.getLeft() )
177  l_receiver.receiveMessage( l_item.getRight() );
178 
179  // clear all messages, that are received
180  l_item.getRight().clear();
181  }
182 
183  m_communicator.send( CCommon.toJson( m_messageflow ) );
184  }

192  public final synchronized void unregister( final CPath p_path, final IParticipant p_receiver )
193  {
194  if ( ( p_path == null ) || ( p_path.isEmpty() ) || ( p_receiver == null ) || ( !m_root.pathExist( p_path ) ) )
195  {
196  CLogger.error( CCommon.getResourceString( this, "unregister", p_receiver, p_path ) );
197  return;
198  }
199 
200  final CTreeNode<Pair<Set<IParticipant>, Set<IMessage>>> l_node = m_root.getNode( p_path );
201  l_node.getData().getLeft().remove( p_receiver );
202 
203  CLogger.info( CCommon.getResourceString( this, "unregistered", p_receiver, p_path ) );
204 
205  for ( final IActionListener l_item : m_listener )
206  l_item.onUnregister( p_path, p_receiver );
207  }

214  */
215  private void incrementMessageFlow( final CPath p_row, final CPath p_colum )
216  {
217  m_messageflow.set(
218  p_row, p_colum, ( m_messageflow.exist( p_row, p_colum ) ? m_messageflow.get(
219  p_row, p_colum
220  ) : new Integer( -1 ) ) + 1
221  );
222  }
