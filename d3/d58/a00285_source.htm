72  @Override
73  public double calculate( final T p_first, final T p_second )
74  {
75  // equal objects create zero value
76  if ( p_first.equals( p_second ) )
77  return 0;
78 
79  final Set<ILiteral<?>> l_firstLiterals = new HashSet<>();
80  final Set<ILiteral<?>> l_secondLiterals = new HashSet<>();
81 
82  // if no path elements are set, we use all
83  if ( m_paths.isEmpty() )
84  {
85  l_firstLiterals.addAll( p_first.getBeliefBase().getLiterals().values() );
86  l_secondLiterals.addAll( p_second.getBeliefBase().getLiterals().values() );
87  }
88  else
89  for ( final CPath l_path : m_paths )
90  {
91  l_firstLiterals.addAll( p_first.getBeliefBase().getLiterals( l_path ).values() );
92  l_secondLiterals.addAll( p_second.getBeliefBase().getLiterals( l_path ).values() );
93  }
94 
95 
96  // get union
97  final Set<ILiteral<?>> l_aggregate = new HashSet<ILiteral<?>>()
98  {{
99  addAll( l_firstLiterals );
100  addAll( l_secondLiterals );
101  }};
102 
103  // difference of contradiction is the sum of difference of contradictions on each belief-base (closed-world-assumption)
104  return new Double( ( ( l_aggregate.size() - l_firstLiterals.size() ) + ( l_aggregate.size() - l_secondLiterals.size() ) ) );
105  }
106 
107 }
