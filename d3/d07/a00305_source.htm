73  @Override
74  public double calculate( final T p_first, final T p_second )
75  {
76  // equal objects create zero value
77  if ( p_first.equals( p_second ) )
78  return 0;
79 
80  // collect all literals within specified paths
81  final Set<ILiteral<?>> l_firstLiterals = new HashSet<>();
82  final Set<ILiteral<?>> l_secondLiterals = new HashSet<>();
83 
84 
85  // if no path elements are set, we use all
86  if ( m_paths.isEmpty() )
87  {
88  l_firstLiterals.addAll( p_first.getBeliefBase().getLiterals().values() );
89  l_secondLiterals.addAll( p_second.getBeliefBase().getLiterals().values() );
90  }
91  else
92  for ( final CPath l_path : m_paths )
93  {
94  l_firstLiterals.addAll( p_first.getBeliefBase().getLiterals( l_path ).values() );
95  l_secondLiterals.addAll( p_second.getBeliefBase().getLiterals( l_path ).values() );
96  }
97 
98  // get size of union
99  final Set<ILiteral<?>> l_set = new HashSet<ILiteral<?>>()
100  {{
101  addAll( l_firstLiterals );
102  addAll( l_secondLiterals );
103  }};
104  final int l_unionSize = l_set.size();
105 
106  // get size of intersection
107  l_set.retainAll( l_firstLiterals );
108  l_set.retainAll( l_secondLiterals );
109  final int l_intersectionSize = l_set.size();
110 
111  // return distance
112  return new Double(
113  ( ( l_unionSize - l_firstLiterals.size() ) +
114  ( l_unionSize - l_secondLiterals.size() ) ) * l_unionSize / l_intersectionSize
115  );
116  }
117 }
