98  /**
99  * checks the syntax of an agent
100  *
101  * @param p_agentname agent name
102  * @note should throw exception on syntax error
103  * @warning exception is not thrown on parsing error, the error is only send to log (see "Agent.parseAS" method), so we pass the log message
104  * to a variable, check the parsing result and the variable content for throwing an own exception
105  */
106  public static void checkAgentFileSyntax( final String p_agentname )
107  {
108  try
109  {
110  // initialize agent manually to modify the internal agent structure (with reflection and pass the
111  // log message to an own logger to get the parsing error messages). The logger on the agent class
112  // is a member variable on the parser class a static variable
113  final CLogger l_logger = new CLogger();
114 
115  final Agent l_agent = new Agent();
116  CReflection.getClassField( Agent.class, "logger" ).getSetter().invokeWithArguments( l_agent, l_logger );
117  // create internal actions map - reset the map and overwrite not useable actions with placeholder
118  CReflection.getClassField( l_agent.getClass(), "internalActions" ).getSetter().invoke( l_agent, INTERNALACTION );
119 
120  final as2j l_parser = new as2j( FileUtils.openInputStream( getAgentFile( p_agentname ) ) );
121  CReflection.getClassField( as2j.class, "logger" ).getSetter().invokeWithArguments( l_logger );
122 
123  // performtemplate agent initalizing and parsing
124  l_agent.initAg();
125  l_parser.agent( l_agent );
126 
127  // on log message wie report an error an create linebreaks for later splitting
128  if ( !l_logger.get().isEmpty() )
129  throw new IllegalStateException( StringUtils.join( l_logger.get(), "\n" ) );
130 
131  }
132  catch ( final Exception l_exception )
133  {
134  throw new IllegalStateException( CCommon.getResourceString( IEnvironment.class, "syntaxerror", p_agentname, "\n\n" + l_exception.getMessage() ) );
135  }
136  catch ( final Throwable l_throwable )
137  {
138  throw new IllegalStateException( CCommon.getResourceString( IEnvironment.class, "syntaxerror", p_agentname, "\n\n" + l_throwable.getMessage() ) );
139  }
140 
141  }

143  /**
144  * creates an agent filename on an agent name
145  *
146  * @param p_agentname agent name
147  * @return file object
148  *
149  * @throws IOException throws IO exception on file creating error
150  */
151  public static File createAgentFile( final String p_agentname ) throws IOException
152  {
153  if ( ( p_agentname == null ) || ( p_agentname.isEmpty() ) )
154  throw new IllegalArgumentException( CCommon.getResourceString( IEnvironment.class, "aslempty" ) );
155 
156  final File l_asl = CConfiguration.getInstance().getLocation( "mas", p_agentname + c_filesuffix );
157  if ( l_asl.exists() )
158  throw new IllegalStateException( CCommon.getResourceString( IEnvironment.class, "aslexist" ) );
159 
160  l_asl.createNewFile();
161  return l_asl;
162  }

164  /**

172  public static File getAgentFile( final String p_agentname )
173  {
174  if ( ( p_agentname == null ) || ( p_agentname.isEmpty() ) )
175  throw new IllegalArgumentException( CCommon.getResourceString( IEnvironment.class, "aslempty" ) );

177  return CConfiguration.getInstance().getLocation( "mas", p_agentname.endsWith( c_filesuffix ) ? p_agentname : p_agentname + c_filesuffix );
