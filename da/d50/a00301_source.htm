89  {

91  m_method = p_method.getHandle();

99  @Override
100  @SuppressWarnings( "unchecked" )
101  public final NanoHTTPD.Response get( final NanoHTTPD.IHTTPSession p_session ) throws Throwable
102  {
103  // parse data - must be called otherwise an time-out exception is thrown
104  if ( NanoHTTPD.Method.PUT.equals( p_session.getMethod() ) || NanoHTTPD.Method.POST.equals( p_session.getMethod() ) )
105  p_session.parseBody( null );
106 
107  try
108  {
109  // invoke method
110  final Object l_return;
111  switch ( m_arguments )
112  {
113  case 0:
114  l_return = m_method.invoke( m_object );
115  break;
116  case 1:
117  l_return = m_method.invoke( m_object, this.convertMap( p_session.getParms() ) );
118  break;
119  case 2:
120  l_return = m_method.invoke( m_object, this.convertMap( p_session.getParms() ), p_session.getHeaders() );
121  break;
122  default:
123  throw new IllegalArgumentException( CCommon.getResourceString( this, "argumentnumber", m_arguments ) );
124  }
125 
126  return new NanoHTTPD.Response( NanoHTTPD.Response.Status.OK, c_mimetype, CCommon.toJson( l_return ) );
127  }
128  catch ( final Exception l_exception )
129  {
130  CLogger.error( l_exception );
131  return new NanoHTTPD.Response(
132  NanoHTTPD.Response.Status.INTERNAL_ERROR, c_mimetype, CCommon.toJson(
133  CCommon.getMap(
134  "error", l_exception.getMessage()
135  )
136  )
137  );
138  }
139  }

168  /**
169  * converts the string-string map into a string-object map to create map-in-map structure
170  *
171  * @param p_input input map
172  * @return string-object map
173  */
174  private Map<String, Object> convertMap( final Map<String, String> p_input )
175  {
176  final Map<String, Object> l_return = new HashMap<>();
177  for ( final Map.Entry<String, String> l_item : p_input.entrySet() )
178  this.splitKeyValues( l_item.getKey().replace( "]", "" ).split( "\\[" ), 0, l_item.getValue(), l_return );
179  return l_return;
180  }
181 
182  /**
183  * split the map into value- and map-types
184  *
185  * @param p_key array with key (separater is "[")
186  * @param p_keyindex current index of the key
187  * @param p_value value
188  * @param p_map return map
189  */
190  private void splitKeyValues( final String[] p_key, final int p_keyindex, final String p_value, final Map<String, Object> p_map )
191  {
192  if ( p_key.length == p_keyindex + 1 )
193  {
194  p_map.put(
195  p_key[p_keyindex], CCommon.convertValue(
196  p_value, new Class[]{Boolean.class, Byte.class, Short.class, Integer.class, Long.class, Float.class, Double.class, Character.class},
197  c_parser
198  )
199  );
200  return;
201  }
202 
203  if ( !p_map.containsKey( p_key[p_keyindex] ) )
204  p_map.put( p_key[p_keyindex], new HashMap<>() );
205 
206  this.splitKeyValues( p_key, p_keyindex + 1, p_value, (Map) p_map.get( p_key[p_keyindex] ) );
207  }
