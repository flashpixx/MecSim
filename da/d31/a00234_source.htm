505  /**
506  * reads the configuration within the directory
507  *
508  * @return boolean on correct read
509  */
510  public boolean read()
511  {
512  final File l_config = this.getLocation( "root", c_filename );
513  CLogger.info( CCommon.getResourceString( this, "read", l_config ) );
514 
515  try
516  {
517  // read Json
518  this.createDirectories();
519  final CNameHashMap l_input = new CNameHashMap( CCommon.fromJson( FileUtils.readFileToString( l_config, "utf-8" ) ) );
520  if ( !l_input.<Boolean>get( "reset" ) )
521  this.setConfiguration( l_input );
522  }
523  catch ( final IllegalArgumentException | ClassNotFoundException | NullPointerException l_exception )
524  {
525  CLogger.error( l_exception.getMessage() );
526  return false;
527  }
528  catch ( final IOException l_exception )
529  {
530  CLogger.error( l_exception );
531  }
532 
533  // append all Jar files to the classpath of the system class loader
534  try
535  {
536  final URLClassLoader l_classloader = (URLClassLoader) ClassLoader.getSystemClassLoader();
537  for ( final String l_jar : m_location.get( "jar" ).list( new WildcardFileFilter( "*.jar" ) ) )
538  {
539  final URL l_fqnjar = CCommon.getResourceURL( m_location.get( "jar" ) + File.separator + l_jar );
540  CReflection.getClassMethod( l_classloader.getClass(), "addURL", new Class<?>[]{URL.class} ).getHandle().invoke(
541  l_classloader, l_fqnjar
542  );
543  CLogger.info( CCommon.getResourceString( this, "jarload", l_fqnjar ) );
544  }
545  }
546  catch ( final Throwable l_throwable )
547  {
548  CLogger.error( l_throwable );
549  return false;
550  }
551 
552  // extract MAS files to the home directory
553  if ( m_configuration.get( "extractmasexamples" ) )
554  for ( final String l_mas : c_masfiles )
555  try
556  {
557  FileUtils.copyURLToFile( CCommon.getResourceURL( "mas/" + l_mas ), new File( m_location.get( "mas" ) + File.separator + l_mas ) );
558  }
559  catch ( final IOException l_exception )
560  {
561  CLogger.error( l_exception );
562  return false;
563  }
564  m_configuration.set( "extractmasexamples", false );
565 
566  // extract external files
567  for ( final String l_file : c_externalfiles )
568  try
569  {
570  FileUtils.copyURLToFile( CCommon.getResourceURL( "external/" + l_file ), new File( m_location.get( "external" ) + File.separator + l_file ) );
571  }
572  catch ( final IOException l_exception )
573  {
574  CLogger.error( l_exception );
575  return false;
576  }
577 
578  return true;
579  }

653  /**
654  * sets the configuration values with semantic check
655  *
656  * @param p_input input map
657  */
658  private void setConfiguration( final CNameHashMap p_input ) throws IOException, ClassNotFoundException, IllegalArgumentException
659  {
660  // convert special data into individual types
661  if ( p_input.traverseContainsKey( "ui/geoposition" ) )
662  p_input.set(
663  "ui/geoposition", new GeoPosition(
664  p_input.<Double>get( "ui/geoposition/latitude" ), p_input.<Double>get( "ui/geoposition/longitude" )
665  )
666  );
667 
668  // check allowed values - traverse default map and transfer values if type is equal - need a local copy of the map for traversing
669  final Set<String> l_errors = new HashSet<>();
670  for ( final Map.Entry<CPath, Object> l_item : m_configuration )
671  if ( p_input.traverseContainsKey( l_item.getKey() ) )
672  {
673  final Object l_data = p_input.get( l_item.getKey() );
674  if ( l_data == null )
675  {
676  m_configuration.set( l_item.getKey(), null );
677  continue;
678  }
679 
680  // performtemplate value checks
681  boolean l_valid = true;
682  if ( m_configurationchecks.containsKey( l_item.getKey().toString() ) )
683  for ( final ICheck l_check : m_configurationchecks.get( l_item.getKey().toString() ) )
684  if ( l_check.isWrong( l_data ) )
685  {
686  l_errors.add( CCommon.getResourceString( this, "valuecheck", l_item.getKey(), l_data, l_check ) );
687  l_valid = false;
688  break;
689  }
690 
691 
692  if ( l_valid )
693  m_configuration.set( l_item.getKey(), l_data );
694  }
695 
696 
697  if ( !l_errors.isEmpty() )
698  throw new IllegalArgumentException( StringUtils.join( l_errors, "\n" ) );
699  }
