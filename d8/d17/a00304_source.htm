76 
77  m_object = p_object;

80 
81  // create heartbeat
82  if ( p_heartbeat < 1 )
83  m_heartbeat = null;
84  else
85  {
86  m_heartbeat = new Timer();
87  m_heartbeat.schedule(
88  new TimerTask()
89  {
90  @Override
91  public void run()
92  {
93  try
94  {
95  CWebSocket.this.ping( "ping".getBytes() );
96  }
97  catch ( final IOException l_exception )
98  {
99  CLogger.error( l_exception );
100  m_heartbeat.cancel();
101  m_heartbeat.purge();
102  }
103  }
104  }, p_heartbeat * c_millisecond, p_heartbeat * c_millisecond
105  );
106  }
107 
108  this.invokeMethod( EAction.Open, null );
109  }
