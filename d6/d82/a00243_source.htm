110  @Override
111  public double calcWeight( final EdgeIteratorState p_edge, final boolean p_reverse, final int p_nextprevedgeid )
112  {
113  if ( this.contains( p_edge.getEdge() ) )
114  return Double.POSITIVE_INFINITY;
115 
116  final double l_speed = p_reverse ? m_encoder.getReverseSpeed( p_edge.getFlags() ) : m_encoder.getSpeed( p_edge.getFlags() );
117  return l_speed == 0 ? Double.POSITIVE_INFINITY : p_edge.getDistance() / l_speed;
118  }

199  final CEdge<?, ?> l_edge = m_graph.getEdge( m_graph.getEdgeIterator( p_edge ) );
200  if ( l_edge == null )
201  return;
202 
203  final Point2D l_start = p_viewer.convertGeoPositionToPoint( l_edge.getGeoPositions( 0 ) );
204  final Point2D l_end = p_viewer.convertGeoPositionToPoint( l_edge.getGeoPositions( -1 ) );
205 
206  p_graphic.drawLine( (int) l_start.getX(), (int) l_start.getY(), (int) l_end.getX(), (int) l_end.getY() );
207  }
