106  {
107  Field l_field = null;

359  /**
360  * scans the defined packages
361  */
362  public void scan()
363  {
364  if ( m_filterdefinition.equals( EFilter.BlackList ) )
365  throw new IllegalStateException( CCommon.getResourceString( this, "blacklist" ) );
366 
367  final List<ClassLoader> l_classloader = new LinkedList<ClassLoader>()
368  {{
369  add( ClasspathHelper.contextClassLoader() );
370  add( ClasspathHelper.staticClassLoader() );
371  }};
372 
373  for ( final String l_item : m_filter )
374  {
375  final Reflections l_reflection = new Reflections( new ConfigurationBuilder()
376  .setScanners( new SubTypesScanner( false ), new ResourcesScanner() )
377  .setUrls( ClasspathHelper
378  .forClassLoader( l_classloader.toArray( new ClassLoader[0] ) ) )
379  .filterInputsBy( new FilterBuilder().include( FilterBuilder.prefix( l_item ) ) ) );
380 
381  for ( final Class<?> l_class : l_reflection.getSubTypesOf( Object.class ) )
382  if ( !l_class.getSimpleName().isEmpty() )
383  m_classname.put( l_class.getSimpleName(), l_class );
384  }

710  public final CMethod get( final String p_methodname, final Class<?>[] p_arguments ) throws IllegalAccessException

720  throw new IllegalAccessException( CCommon.getResourceString( this, "access", p_methodname ) );
