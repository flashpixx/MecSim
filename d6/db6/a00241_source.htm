377  protected class CCollectorFundamentalDiagram extends IDatabase.CWorker

394  @Override
395  public void step( final int p_currentstep, final ILayer p_layer ) throws Exception
396  {
397  CEvaluationStore.this.updateInsertStatement( m_statement, p_currentstep );
398  final CCarLayer l_cars = CSimulation.getInstance().getWorld().<CCarLayer>getTyped( "Cars" );
399 
400  l_cars.getGraph().getEdgeCollection().parallelStream().filter( i -> i.getNumberOfObjects() > 0 ).forEach(
401  ( l_edge ) ->
402  {
403  final double l_carcount = l_edge.getNumberOfObjects();
404 
405  // calculate sum over all speeds for averages
406  final double l_speedsum = l_edge.getCellList().stream().filter( i -> i != null ).mapToDouble( i -> i.getCurrentSpeed() ).sum();
407  final double l_speedsuminvert = l_edge.getCellList().stream().filter( i -> i != null ).mapToDouble( i -> 1.0 / i.getCurrentSpeed() )
408  .sum();
409 
410  try
411  {
412  m_statement.setObject( 5, l_edge.getEdgeID(), Types.BIGINT );
413 
414  // traffic power = cars / time (in seconds)
415  m_statement.setDouble( 6, l_carcount / l_cars.getUnitConvert().getTime() );
416  // traffic density = cars / distance (in meter)
417  m_statement.setDouble( 7, l_carcount / l_edge.getDistance() );
418 
419  // harmonic mean speed = sum car speed / sum ( 1 / car speed )
420  m_statement.setDouble( 8, l_speedsum / l_speedsuminvert );
421  // average speed
422  m_statement.setDouble( 9, l_speedsum / l_carcount );
423 
424  m_statement.execute();
425 
426  }
427  catch ( final SQLException l_exception )
428  {
429  CLogger.error( l_exception );
430  }
431 
432  }
433  );
434  }
435  }
