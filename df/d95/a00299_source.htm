64  {
65  if ( ( p_uri == null ) || ( p_uri.isEmpty() ) || ( p_uri.endsWith( c_seperator ) ) )
66  throw new IllegalArgumentException( CCommon.getResourceString( this, "trailingslashempty", p_uri ) );
67  if ( p_file == null )
68  throw new IllegalArgumentException( CCommon.getResourceString( this, "fileisnull", p_uri ) );
69 
70  m_file = p_file;
71  m_uri = p_uri.startsWith( c_seperator ) ? p_uri.replaceAll( c_allowchar, "" ) : c_seperator + p_uri.replaceAll( c_allowchar, "" );
72 
73  CLogger.info( p_uri );
74  }
