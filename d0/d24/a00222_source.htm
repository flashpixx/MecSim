64  @Override
65  protected void updating()
66  {
67  // iterate over all binded objects
68  for ( final Map.Entry<String, Pair<Object, Map<String, CReflection.CGetSet>>> l_item : m_bind.entrySet() )
69 
70  // iterate over all object fields
71  for ( final Map.Entry<String, CReflection.CGetSet> l_fieldref : l_item.getValue().getRight().entrySet() )
72  try
73  {
74  // invoke / call the getter of the object field - field name will be the belief name, return value
75  // of the getter invoke call is set for the belief value
76  // replace separator to avoid path splitting
77  String l_literalname = l_fieldref.getKey();
78  for ( final String l_replace : m_literalreplace )
79  l_literalname = l_literalname.replace( l_replace, "" );
80 
81  final Literal l_literal = this.add(
82  CCommon.getLiteral(
83  l_literalname, l_fieldref.getValue().getGetter().invoke(
84  l_item.getValue().getLeft()
85  )
86  )
87  );
88 
89  // add the annotation to the belief and push it to the main list for reading later (within the agent)
90  l_literal.addAnnot( CCommon.getLiteral( "source", ASSyntax.createAtom( l_item.getKey() ) ) );
91  }
92  catch ( final Exception l_exception )
93  {
94  CLogger.error(
95  de.tu_clausthal.in.mec.common.CCommon.getResourceString(
96  this, "getter", l_item.getKey(), l_fieldref.getKey(), l_exception.getMessage()
97  )
98  );
99  }
100  catch ( final Throwable l_throwable )
101  {
102  CLogger.error(
103  de.tu_clausthal.in.mec.common.CCommon.getResourceString(
104  this, "getter", l_item.getKey(), l_fieldref.getKey(), l_throwable.getMessage()
105  )
106  );
107  }
108  }
