53  /**
54  * creates a list with the layer objects depends on the ordering value
55  *
56  * @return list of layer
57  */
58  @SuppressWarnings( value = "unchecked" )
59  public final List<ILayer> getOrderedLayer()
60  {
61 
62  final MultiMap<Integer, ILayer> l_order = new MultiValueMap<>();
63  for ( final ILayer l_layer : m_layer.values() )
64  l_order.put( l_layer.getCalculationIndex(), l_layer );
65 
66  // get key values and sort it
67  final Object[] l_sortkeys = l_order.keySet().toArray();
68  Arrays.sort( l_sortkeys );
69 
70  // build the list of the layer
71  final List<ILayer> l_list = new LinkedList<>();
72  for ( final Object l_key : l_sortkeys )
73  l_list.addAll( (Collection<ILayer>) l_order.get( l_key ) );
74 
75  return l_list;
76  }
