106  /**
107  * get a object name, depend on simulation data
108  *
109  * @param p_input input string
110  * @param p_object object for object hash
111  * @return string with name
112  *
113  * @note %hash% with the object hash or 0, %step% with the current simulation step, %rand% with a random integer value, %inc% increment value
114  */
115  public final String generateObjectName( final String p_input, final Object p_object )
116  {
117  String l_return = new String( p_input );
118 
119  l_return = l_return.replace( "%hash%", new Integer( p_object != null ? p_object.hashCode() : 0 ).toString() );
120  l_return = l_return.replace( "%step%", new Integer( m_mainloop.getSimulationstep() ).toString() );
121  if ( l_return.contains( "%rand%" ) )
122  l_return = l_return.replace( "%rand%", new Integer( m_random.nextInt() ).toString() );
123  if ( l_return.contains( "%inc%" ) )
124  l_return = l_return.replace( "%inc%", new Long( m_objectcounter.getAndIncrement() ).toString() );
125 
126  return l_return;
127  }

197  /**
198  * runs the simulation for n steps
199  *
200  * @param p_steps number of steps
201  * @throws InterruptedException throws the exception on thread startup
202  */
203  public void start( final int p_steps ) throws InterruptedException
204  {
205  if ( this.isRunning() )
206  throw new IllegalStateException( CCommon.getResourceString( this, "running" ) );
207  this.threadStartUp();
208 
209  CLogger.info( CCommon.getResourceString( this, "startsteps", p_steps ) );
210 
211  // performtemplate thread and wait until thread is finished
212  this.callLayerStart();
213  m_mainloop.resume( p_steps );
214  m_mainloopthread.join();
215  m_mainloop.stop();
216 
217  m_mainloopthread = null;
218  m_runs++;
219  }

449  /**

452  public class CStorage
453  {
454  /**
455  * map with (lower-case) names and elements
456  */
457  private final Map<String, Object> m_elements = new HashMap<>();
458 
459  /**
460  * adds an element
461  *
462  * @param p_name name of the object
463  * @param p_object object
464  * @warning existing elements are not allowed and throws an exception
465  */
466  public final void add( final String p_name, final Object p_object )
467  {
468  if ( m_elements.containsKey( p_name.toLowerCase() ) )
469  throw new IllegalArgumentException( CCommon.getResourceString( this, "exists", p_name ) );
470 
471  m_elements.put( p_name.toLowerCase(), p_object );
472  }
473 
474  /**
475  * check the map has existing elements
476  *
477  * @return existence flag
478  */
479  public final boolean exists()
480  {
481  return !m_elements.isEmpty();
482  }
483 
484  /**
485  * check if an object exists
486  *
487  * @param p_name name of the object
488  * @return existence flag

490  public final boolean exists( final String p_name )
491  {
492  return m_elements.containsKey( p_name.toLowerCase() );
493  }

500  *

502  */

504  {
