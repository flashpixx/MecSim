65  @Override
66  public ILiteral<Literal> clone( final CPath p_prefix )
67  {
68  // don't use setNegated, because setNegated returns the new literal object, so
69  // we generate a literal directly with the correct negation operator (first parameter
70  // is the "positiv" definition of negated, so we need to invert the data)
71  // Literal base class implements setter with log messages and does not declare it with abstract
72  final Literal l_literal = ASSyntax.createLiteral( !m_literal.negated(), p_prefix.append( m_functor.get() ).toString() );
73  l_literal.setAnnots( m_literal.getAnnots() );
74  l_literal.setTerms( m_literal.getTerms() );
75  return new CLiteral( l_literal );
76  }

79  /**
80  * modifies the interal literal representations
81  *
82  * @param p_literal literal
83  */
84  private void addAdditionals( final Literal p_literal )
85  {
86  if ( p_literal.hasTerm() )
87  for ( final Term l_term : p_literal.getTerms() )
88  m_values.add( CCommon.convertGeneric( l_term ) );
89 
90  if ( p_literal.hasAnnot() )
91  for ( final Term l_term : p_literal.getAnnots() )
92  m_annotations.add( CCommon.convertGeneric( l_term ) );
93  }
