95  @Override
96  public Collection<Pair<GeoPosition, GeoPosition>> getPath()
97  {
98  HashSet<Pair<GeoPosition, GeoPosition>> l_path = new HashSet<>();
99 
100  //falls kein eintrag mehr da ist fahre zu default und breche ab
101 
102  //inital position
103  Map<IWayPoint, MutablePair<Double, Double>> l_currentNode = m_makrovChain.get( this );
104  GeoPosition l_currentPosition = this.getPosition();
105  GeoPosition l_default = new GeoPosition( 51.80135377062704, 10.32871163482666 );
106  GeoPosition l_newPosition;
107 
108  //check for circles and path length
109  for ( int i = 1; i < m_makrovChain.size(); i++ )
110  {
111  double l_random = m_random.nextDouble();
112 
113  // current node has no outgoing edges add a path to default target and return
114  if ( l_currentNode.size() <= 0 )
115  {
116  CLogger.out( l_path.size() );
117  l_path.add( new ImmutablePair<>( l_currentPosition, l_default ) );
118  return l_path;
119  }
120 
121  //calculate new edge
122  for ( final Map.Entry<IWayPoint, MutablePair<Double, Double>> l_node : l_currentNode.entrySet() )
123  {
124  if ( l_random >= l_node.getValue().getRight() )
125  {
126  l_random = l_random - l_node.getValue().getRight();
127  }
128  else
129  {
130  l_newPosition = l_node.getKey().getPosition();
131  l_path.add( new ImmutablePair<>( l_currentPosition, l_newPosition ) );
132  l_currentNode = m_makrovChain.get( l_node.getKey() );
133  l_currentPosition = l_newPosition;
134  break;
135  }
136  }
137  }
138 
139  return l_path;
140  }

148  /**
149  * generic class to create a markov chain
150  * this class also provides a relative and absolute weighting
151  *
152  * @param <T>
153  */
154  public static class CMakrovChain<T> extends HashMap<T, Map<T, MutablePair<Double, Double>>>
155  {
156  /**
157  * add edge to makrov chain
158  *
159  * @param p_start
160  * @param p_end
161  * @param p_value
162  */
163  public final void addEdge( final T p_start, final T p_end, final double p_value )
164  {
165  if ( p_value < 0 )
166  return;
167 
168  //create node if does not exist
169  if ( !this.containsKey( p_start ) )
170  this.addNode( p_start );
171  if ( !this.containsKey( p_end ) )
172  this.addNode( p_end );
173 
174  this.get( p_start ).put( p_end, new MutablePair<>( p_value, p_value ) );
175  this.updateRelativeWeighting();
176  }

183  public final void addNode( final T p_node )
184  {

243  /**
244  * update all relative weightings
245  */
246  private void updateRelativeWeighting()
247  {
248  for ( final Map<T, MutablePair<Double, Double>> l_edge : this.values() )
249  {
250  //calculate the new sum
251  double l_sum = 0.0;
252  for ( final MutablePair l_pair : l_edge.values() )
253  l_sum += (double) l_pair.left;
254 
255  //update relative weighting
256  for ( final MutablePair l_pair : l_edge.values() )
257  l_pair.right = (double) l_pair.left / l_sum;
258  }
259  }
