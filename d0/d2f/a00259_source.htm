458  /**
459  * returns a mask on the recursive descend
460  *
461  * @param p_path path
462  * @param p_root start / root node
463  * @param p_generator generator object for new masks
464  * @return mask
465  *
466  * @note a path can contains ".." to use the parent object
467  * @tparam Q literal type
468  */
469  private static <Q> IBeliefBaseMask<Q> walk( final CPath p_path, final IBeliefBaseMask<Q> p_root, final IGenerator<Q> p_generator )
470  {
471  if ( ( p_path == null ) || ( p_path.isEmpty() ) )
472  return p_root;
473 
474  // get the next mask (on ".." the parent is returned otherwise the child is used)
475  IBeliefBaseMask<Q> l_mask = "..".equals( p_path.get( 0 ) ) ? p_root.getParent() : p_root.getStorage().getSingleElement( p_path.get( 0 ) );
476 
477  // if a generator is exists and the mask is null, a new mask is created and added to the current
478  if ( ( l_mask == null ) && ( p_generator != null ) && ( !( "..".equals( p_path.get( 0 ) ) ) ) )
479  l_mask = p_root.add( p_generator.createBeliefbase( p_path.get( 0 ) ) );
480 
481  // if mask null an exception is thrown
482  if ( l_mask == null )
483  throw new IllegalArgumentException( CCommon.getResourceString( CMask.class, "pathelementnotfound", p_path.get( 0 ), p_path ) );
484 
485  // recursive descend
486  return walk( p_path.getSubPath( 1 ), l_mask, p_generator );
487  }
