46  /**
47  * agent instantiate function
48  *
49  * @param p_source source of the agent
50  * @param p_task optional tasks which runs over the agenttemplate
51  * @param p_any any object definition
52  * @return agent
53  */
54  public final <N> N instantiate( final File p_source, final ITask<T> p_task, final L... p_any ) throws Exception
55  {
56  final T l_agenttemplate = m_storage.getOrDefault( p_source, this.create( p_source ) );
57  m_storage.putIfAbsent( p_source, l_agenttemplate );
58 
59  if ( p_task != null )
60  p_task.performtemplate( l_agenttemplate );
61 
62  return this.clone( l_agenttemplate, p_any );
63  }
