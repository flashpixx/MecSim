60  /**
61  * ctor
62  *
63  * @param p_startstep start step
64  * @param p_endstep end step
65  * @param p_maxobjects maximum number of created objects
66  */
67  public CConstant( final int p_startstep, final int p_endstep, final int p_maxobjects )
68  {
69  if ( p_startstep >= p_endstep )
70  throw new IllegalArgumentException( CCommon.getResourceString( this, "startbegin " ) );
71 
72  final int l_duration = p_endstep - p_startstep;
73 
74  m_startstep = p_startstep;
75  m_endstep = p_endstep;
76  m_stepobjects = p_maxobjects / BigInteger.valueOf( Math.abs( p_maxobjects ) ).gcd( BigInteger.valueOf( l_duration ) ).intValue();
77 
78  if ( m_stepobjects * l_duration != p_maxobjects )
79  throw new IllegalArgumentException( CCommon.getResourceString( this, "notpossible" ) );
80 
81  m_inspect.put( CCommon.getResourceString( this, "startstep" ), m_startstep );
82  m_inspect.put( CCommon.getResourceString( this, "endstep" ), m_endstep );
83  m_inspect.put( CCommon.getResourceString( this, "objectstep" ), m_stepobjects );
84  }
85 
86 
87  @Override
88  public int getCount( final int p_currentStep )
89  {
90  if ( ( p_currentStep >= m_startstep ) && ( p_currentStep <= m_endstep ) )
91  return m_stepobjects;
92  return 0;
93  }
