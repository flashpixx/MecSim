234  /**
235  * routing painter class
236  */
237  private class CLinePainter implements ActionListener, Painter<JXMapViewer>
238  {
239  /**
240  * alpha value
241  */
242  private float m_alpha = 1;
243  /**
244  * route list with painting structure *
245  */
246  private List<Triple<Pair<GeoPosition, GeoPosition>, Color, Stroke>> m_line;
247  /**
248  * timer
249  */
250  private final Timer m_timer = new Timer( CConfiguration.getInstance().get().<Integer>get( "ui/routepainterdelay" ), this );
251 
252  @Override
253  public void actionPerformed( final ActionEvent p_event )
254  {
255  m_alpha -= 0.01;
256  if ( m_alpha <= 0 )
257  {
258  m_line = null;
259  m_alpha = 0;
260  m_timer.stop();
261  }
262  COSMViewer.this.repaint();
263  }
264 
265  @Override
266  public void paint( final Graphics2D p_graphic, final JXMapViewer p_viewer, final int p_width, final int p_height )
267  {
268  if ( ( m_line == null ) || ( m_line.isEmpty() ) )
269  return;
270 
271  p_graphic.setComposite( AlphaComposite.getInstance( AlphaComposite.SRC_OVER, m_alpha ) );
272  p_graphic.translate( -p_viewer.getViewportBounds().x, -p_viewer.getViewportBounds().y );
273  p_graphic.setRenderingHint( RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON );
274 
275  for ( final Triple<Pair<GeoPosition, GeoPosition>, Color, Stroke> l_item : m_line )
276  {
277  p_graphic.setColor( l_item.getMiddle() );
278  p_graphic.setStroke( l_item.getRight() );
279 
280  final Point2D l_start = p_viewer.getTileFactory().geoToPixel( l_item.getLeft().getLeft(), p_viewer.getZoom() );
281  final Point2D l_end = p_viewer.getTileFactory().geoToPixel( l_item.getLeft().getRight(), p_viewer.getZoom() );
282  p_graphic.drawLine( (int) l_start.getX(), (int) l_start.getY(), (int) l_end.getX(), (int) l_end.getY() );
283  }
284  }
