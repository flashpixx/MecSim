88  /**
89  * ctor create the samples
90  *
91  * @param p_edgestate edge
92  * @param p_cellsize samplling size of a cell
93  */
94  public CEdge( final EdgeIteratorState p_edgestate, final int p_cellsize )
95  {
96  m_edgeid = p_edgestate.getEdge();
97  m_edgedistance = p_edgestate.getDistance();
98 
99 
100  final ArrayList<N> l_initlist = new ArrayList<>();
101  for ( int i = 0; i < (int) Math.ceil(
102  m_edgedistance / p_cellsize
103  ); i++
104  )
105  l_initlist.add( null );
106  m_cells = (N[]) l_initlist.toArray();
107  m_additionalinformation = (T[]) l_initlist.toArray();
108  m_cellgeoposition = new GeoPosition[m_cells.length];
109 
110  // create a spline interpolation for cell sampling of the geoposition
111  // get edge geoposition and convert data in arrays, performtemplate spline interpolation
112  // catch number exceptions and calculate for each cell the geoposition
113  final PointListArray l_list = this.filterPointList( p_edgestate.fetchWayGeometry( 2 ), c_epsilon );
114 
115  try
116  {
117 
118  final UnivariateInterpolator l_interpolator = l_list.size() < 3 ? new LinearInterpolator() : new SplineInterpolator();
119  final UnivariateFunction l_function = l_interpolator.interpolate( l_list.getX(), l_list.getY() );
120  final double l_increment = ( l_list.getX( l_list.size() - 1 ) - l_list.getX( 0 ) ) / m_cells.length;
121  for ( int i = 0; i < m_cells.length; i++ )
122  m_cellgeoposition[i] = new GeoPosition( l_list.getX( 0 ) + i * l_increment, l_function.value( l_list.getX( 0 ) + i * l_increment ) );
123 
124  }
125  catch ( final NonMonotonicSequenceException l_exception )
126  {
127 
128  final double l_xincrement = ( l_list.getX( l_list.size() - 1 ) - l_list.getX( 0 ) ) / m_cells.length;
129  final double l_yincrement = ( l_list.getY( l_list.size() - 1 ) - l_list.getY( 0 ) ) / m_cells.length;
130  for ( int i = 0; i < m_cells.length; i++ )
131  m_cellgeoposition[i] = new GeoPosition( l_list.getX( 0 ) + i * l_xincrement, l_list.getY( 0 ) + i * l_yincrement );
132  }
133  }

423  /**
424  * filter point list to create a list of points which stores monoton increase points
425  *
426  * @param p_input input point list
427  * @param p_epsilon epsilon value of the monotonic increase
428  * @return point list array
429  */
430  private final PointListArray filterPointList( final PointList p_input, final double p_epsilon )
431  {
432  final ArrayList<Double> l_x = new ArrayList<>();
433  final ArrayList<Double> l_y = new ArrayList<>();
434 
435  l_x.add( p_input.getLatitude( 0 ) );
436  l_y.add( p_input.getLongitude( 0 ) );
437 
438  // convert point list to arrays and beware static increase
439  for ( int i = 1; i < p_input.size() - 1; i++ )
440  if ( ( Math.abs( l_x.get( l_x.size() - 1 ) - p_input.getLatitude( i ) ) >= p_epsilon ) && ( Math.abs(
441  l_y.get(
442  l_y.size() - 1
443  ) - p_input.getLongitude( i )
444  ) >= p_epsilon ) )
445  {
446  l_x.add( p_input.getLatitude( i ) );
447  l_y.add( p_input.getLongitude( i ) );
448  }
449 
450  l_x.add( p_input.getLatitude( p_input.size() - 1 ) );
451  l_y.add( p_input.getLongitude( p_input.size() - 1 ) );
452 
453  return new PointListArray( l_x, l_y );
454  }
