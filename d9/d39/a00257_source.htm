167  @Override
168  public final void run()
169  {
170  CLogger.info( CCommon.getResourceString( this, "start" ) );
171 
172  // order of all layer - the order will be read only once
173  // so the thread need not be startup on program initializing
174  final List<ILayer> l_layerorder = CSimulation.getInstance().getWorld().getOrderedLayer();
175  CLogger.info( l_layerorder );
176 
177  while ( !Thread.currentThread().isInterrupted() )
178  {
179 
180  try
181  {
182  // if thread is paused
183  if ( m_pause )
184  {
185  Thread.yield();
186  continue;
187  }
188 
189  // shutdown
190  if ( m_simulationcount >= m_shutdownstep )
191  break;
192 
193  // performtemplate simulation objects
194  this.processLayer( l_layerorder );
195  this.processObjects( l_layerorder );
196 
197  m_simulationcount++;
198  Thread.sleep( m_sleeptime );
199  }
200  catch ( final InterruptedException l_exception )
201  {
202  Thread.currentThread().interrupt();
203  return;
204  }
205  }
206 
207  m_pool.shutdownNow();
208  CLogger.info( CCommon.getResourceString( this, "stop" ) );
209  }

219  /**
220  * invokes all defined data
221  *
222  * @param p_layer layer
223  * @param p_tasksource invoking tasks
224  * @throws InterruptedException is throws on task error
225  */
226  protected final void invokeTasks( final ILayer p_layer, final Collection<ISteppable> p_tasksource ) throws InterruptedException
227  {
228  m_pool.invokeAll(
229  p_tasksource.parallelStream().map(
230  i -> createTask( m_simulationcount, i, p_layer )
231  ).collect( Collectors.toCollection( () -> new LinkedList<>() ) )
232  );
233  }

235  /**
236  * returns a runnable object of the steppable input
237  *
238  * @param p_iteration iteration
239  * @param p_object steppable object
240  * @param p_layer layer
241  * @return runnable object
242  */
243  @SuppressWarnings( "unchecked" )
244  private static Callable<Object> createTask( final int p_iteration, final ISteppable p_object, final ILayer p_layer )
245  {
246  if ( p_object instanceof IVoidSteppable )
247  return new CVoidSteppable( p_iteration, (IVoidSteppable) p_object, p_layer );
248 
249  if ( p_object instanceof IReturnSteppable )
250  return new CReturnSteppable( p_iteration, (IReturnSteppable) p_object, p_layer );
251 
252  throw new IllegalArgumentException( CCommon.getResourceString( CMainLoop.class, "notsteppable" ) );
253  }

256  /**
257  * process layer
258  *
259  * @param p_layer ordered layer list
260  * @throws InterruptedException thrown on thread error
261  */
262  @IBenchmark
263  private void processLayer( final List<ILayer> p_layer ) throws InterruptedException
264  {
265  final Collection<Callable<Object>> l_tasks = new LinkedList<>();
266 
267  l_tasks.add( new CVoidSteppable( m_simulationcount, CSimulation.getInstance().getMessageSystem(), null ) );
268  p_layer.stream().filter( i -> i.isActive() ).map( i -> createTask( m_simulationcount, i, null ) ).forEachOrdered( l_tasks::add );
269 
270  m_pool.invokeAll( l_tasks );
271  }
272 
273  /**
274  * process layer object
275  *
276  * @param p_layer ordered layer list
277  * @note only multi-, evaluate- & network layer can store other objects
278  */
279  @IBenchmark
280  @SuppressWarnings( "unchecked" )
281  private void processObjects( final List<ILayer> p_layer )
282  {
283  p_layer.stream().filter( i -> i.isActive() && ( !( i instanceof ISingleLayer ) ) && ( !( i instanceof ISingleEvaluateLayer ) ) ).forEachOrdered( i -> {
284 
285  try
286  {
287  if ( i instanceof IMultiLayer<?> )
288  this.invokeTasks( i, (IMultiLayer) i );
289 
290  if ( i instanceof IMultiEvaluateLayer<?> )
291  this.invokeTasks( i, (IMultiEvaluateLayer) i );
292  }
293  catch ( final InterruptedException l_exception )
294  {
295  CLogger.error( l_exception );
296  }
297  } );
298  }
