76  /**
77  * hidden event handler
78  */
79  private final EventHandler<WindowEvent> m_popuphiddenevent = new EventHandler<WindowEvent>()
80  {
81  @Override
82  public void handle( final WindowEvent p_event )
83  {
84  p_event.consume();
85 
86  final Node l_node = ( (PopOver) p_event.getSource() ).getContentNode();
87  if ( !m_widget.containsKey( l_node ) )
88  return;
89 
90  m_tabpane.getTabs().add( m_widget.get( l_node ).getValue() );
91  }
92  };
93  /**
94  * tab close handler
95  */
96  private final EventHandler<Event> m_tabcloseevent = new EventHandler<Event>()
97  {
98  @Override
99  public void handle( final Event p_event )
100  {
101  p_event.consume();
102 
103  final Tab l_tab = (Tab) p_event.getSource();
104  if ( !m_widget.containsKey( l_tab.getContent() ) )
105  return;
106 
107  m_widget.get( l_tab.getContent() ).getKey().show( m_stage );
108  }

127  public final void add( final String p_title, final Node p_node )
128  {
129  if ( ( m_widget.containsKey( p_node ) ) || ( m_content.containsKey( p_title ) ) )
130  return;
131 
132  m_content.put( p_title, p_node );

134  }

159  /**

166  private PopOver createPopOver( final String p_title, final Node p_node )
167  {
168  final PopOver l_popover = new PopOver( p_node );
169

248

250  l_root.setCenter( m_tabpane );
251  p_stage.setScene(
252  new Scene(
253  l_root, Math.min(
254  Screen.getPrimary().getVisualBounds().getWidth() * 0.9, CConfiguration.getInstance().get().<Number>get( "ui/windowwidth" ).doubleValue()
255  ), Math.min(
256  Screen.getPrimary().getVisualBounds().getHeight() * 0.9, CConfiguration.getInstance().get().<Number>get( "ui/windowheight" )
257  .doubleValue()
258  )
259  )
260  );
261  p_stage.sizeToScene();
262  p_stage.show();
263  }
