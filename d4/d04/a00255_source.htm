362  /**
363  * creates a full log item
364  *
365  * @param p_status status name
366  * @param p_add additional log data
367  */
368  private static String createLogData( final Level p_status, final Object p_add )
369  {
370  // get invoker position (index depends on static main function)
371  int l_invokerindex;
372  try
373  {
374  getInvokingMethodNameFqn( 4 );
375  l_invokerindex = 4;
376  }
377  catch ( final ArrayIndexOutOfBoundsException l_exception )
378  {
379  l_invokerindex = 3;
380  }
381 
382  final String l_sep = StringUtils.repeat( " ", 5 );
383  final StringBuffer l_str = new StringBuffer();
384 
385  l_str.append( l_sep ).
386  append( padCut( "status [" + p_status + "]", ' ', 15 ) ).
387  append( l_sep ).
388  append( padCut( "thread [" + Thread.currentThread() + "]", ' ', 100 ) ).
389  append( l_sep ).
390  append( padCut( "invoker [" + getInvokingMethodNameFqn( l_invokerindex ) + "]", ' ', 100 ) ).
391  append( l_sep ).
392  append( padCut( "method [" + getCurrentMethodNameFqn( 3 ) + "]", ' ', 100 ) ).
393  append( l_sep ).
394  append( padCut( "line no [" + getCurrentLineNumber( 3 ) + "]", ' ', 25 ) );
395 
396 
397  String l_add = "";
398  if ( p_add != null )
399  {
400  if ( p_add instanceof Collection )
401  {
402  l_add = "[ ";
403  for ( Object l_item : (Collection) p_add )
404  l_add += l_item.toString().trim() + " ";
405  l_add += "]";
406  }
407  else
408  l_add = p_add.toString().trim();
409  }
410 
411  if ( !l_add.isEmpty() )
412  {
413  l_str.append( l_sep ).
414  append( l_add.replace( "\n", " " ).replace( "\t", " " ).replace( "\r", "" ).trim() );
415  }
416 
417  return l_str.toString().trim();
418  }

496  /**
497  * pad / cut string of a define length
498  *
499  * @param p_input input string
500  * @param p_filler fill character
501  * @param p_length max string length
502  * @return modified string
503  */
504  private static String padCut( final String p_input, final char p_filler, final int p_length )
505  {
506  if ( p_length < 1 )
507  return p_input;
508  if ( p_input.length() < p_length )
509  return p_input + StringUtils.repeat( p_filler, p_length - p_input.length() );
510 
511  return p_input.substring( 0, p_length );
512  }
