185  l_graphics.setRenderingHint( RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BILINEAR );

197  @Override
198  public final void onClick( final MouseEvent p_event, final JXMapViewer p_viewer )
199  {
200  if ( ( !m_visible ) || ( m_position == null ) )
201  return;
202 
203  final int l_zoom = this.iconsize( p_viewer );
204  final Point2D l_point = p_viewer.getTileFactory().geoToPixel( m_position, p_viewer.getZoom() );
205  final Ellipse2D l_circle = new Ellipse2D.Double(
206  l_point.getX() - p_viewer.getViewportBounds().getX() - l_zoom,
207  l_point.getY() - p_viewer.getViewportBounds().getY() - l_zoom,
208  l_zoom * 2, l_zoom * 2
209  );
210 
211  if ( l_circle.contains( p_event.getX(), p_event.getY() ) )
212  CSimulation.getInstance().getStorage().<CInspector>get( "inspector" ).set( this );
213  }

215  @Override
216  public final void paint( final Graphics2D p_graphic, final COSMViewer p_viewer, final int p_width, final int p_height )
217  {
218  if ( !m_visible )
219  return;
220 
221  //if the zoom change calculate the new scaled image
222  if ( p_viewer.getZoom() != m_lastZoom )
223  m_scaledimage = this.getScaledImage(
224  m_initializeimage, (int) ( m_initializeimage.getWidth() * this.iconscale( p_viewer ) ),
225  (int) ( m_initializeimage.getHeight() * this.iconscale( p_viewer ) )
226  );
227 
228  final Point2D l_point = p_viewer.getTileFactory().geoToPixel( m_position, p_viewer.getZoom() );
229  p_graphic.drawImage( m_scaledimage, (int) l_point.getX() - m_scaledimage.getWidth() / 2, (int) l_point.getY() - m_scaledimage.getHeight(), null );
230  }

242  /**
243  * creates an image with a specific scale
244  *
245  * @param p_width image width
246  * @param p_height image height
247  * @param p_color color of the waypoint
248  * @return image
249  */
250  private BufferedImage initializeImage( final int p_width, final int p_height, final Color p_color )
251  {
252  try
253  {
254  final BufferedImage l_image = this.getScaledImage(
255  ImageIO.read( DefaultWaypointRenderer.class.getResource( "/images/standard_waypoint.png" ) ), p_width, p_height
256  );
257  for ( int i = 0; i < l_image.getHeight(); i++ )
258  for ( int j = 0; j < l_image.getWidth(); j++ )
259  {
260  final Color l_color = new Color( l_image.getRGB( j, i ) );
261  if ( l_color.getBlue() > 0 )
262  l_image.setRGB( j, i, p_color.getRGB() );
263  }
264 
265  return l_image;
266  }
267  catch ( final Exception l_exception )
268  {
269  CLogger.error( l_exception );
270  }
271 
272  return null;
273  }
