326  @Override
327  @IBenchmark
328  @CMethodFilter.CAgent( bind = false )
329  public void step( final int p_currentstep, final ILayer p_layer ) throws Exception
330  {
331 
332  // if the car is at the end
333  if ( this.hasEndReached() )
334  return;
335 
336  // if the car reaches the end
337  int l_speed = m_layer.getUnitConvert().getSpeedToCell( this.getCurrentSpeed() );
338  if ( m_routeindex + l_speed >= m_route.size() )
339  {
340  m_endreached = true;
341  l_speed = m_route.size() - m_routeindex - 1;
342  }
343 
344  // if the route index equal to zero, push it car on the first item or wait until it is free
345  if ( m_routeindex == 0 )
346  {
347 
348  if ( !m_layer.getGraph().getEdge( m_route.get( l_speed ).getLeft() ).isEmpty( m_route.get( l_speed ).getRight().intValue() ) )
349  return;
350 
351  try
352  {
353  m_layer.getGraph().getEdge( m_route.get( l_speed ).getLeft() ).setObject( this, m_route.get( l_speed ).getRight().intValue() );
354  m_routeindex += l_speed;
355  }
356  catch ( final IllegalAccessException l_exception )
357  {
358  }
359 
360  }
361  else
362  {
363 
364  try
365  {
366  m_layer.getGraph().getEdge( m_route.get( m_routeindex ).getLeft() ).removeObject( this );
367  m_layer.getGraph().getEdge( m_route.get( m_routeindex + l_speed ).getLeft() ).setObject(
368  this, m_route.get( m_routeindex + l_speed ).getRight()
369  );
370  m_routeindex += l_speed;
371  }
372  catch ( final IllegalAccessException l_exception )
373  {
374  m_layer.getGraph().getEdge( m_route.get( m_routeindex ).getLeft() ).removeObject( this );
375  }
376 
377  }
378 
379  }
