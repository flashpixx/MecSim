348  protected final CPath getFQN( final CTreeNode<T> p_node )

358  /**
359  * returns the node
360  *
361  * @param p_node current node
362  * @param p_path path
363  * @param p_index current path index
364  * @return node
365  */
366  protected final CTreeNode<T> getNode( final CTreeNode<T> p_node, final CPath p_path, final int p_index )
367  {
368  if ( p_index < p_path.size() )
369  if ( p_node.m_childs.containsKey( p_path.get( p_index ) ) )
370  return p_node.getNode( p_node.m_childs.get( p_path.get( p_index ) ), p_path, p_index + 1 );
371  else
372  return p_node.getNode( new CTreeNode<T>( p_path.get( p_index ), p_node ), p_path, p_index + 1 );
373 
374  return p_node;
375  }

377  /**

379  *

381  * @param p_node start node
382  * @param p_set return set
383  */
384  protected final void getTree( final CPath p_path, final CTreeNode<T> p_node, final Set<CPath> p_set )
385  {
386  if ( !p_path.isEmpty() )
387  p_set.add( p_path );
388 
389  for ( final Map.Entry<String, CTreeNode<T>> l_item : m_childs.entrySet() )
390  {
391  final CPath l_path = new CPath( p_path );
392  l_path.pushback( l_item.getKey() );
393  l_item.getValue().getTree( l_path, l_item.getValue(), p_set );
394  }
395  }
