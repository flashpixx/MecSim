155  /**
156  * registerObject an object for the UI
157  *
158  * @param p_object object, all methods with the name "ui_" are registered
159  * @note a class / object need to implemente methods with the prefix "web_static_" or "web_dynamic_", which is bind
160  * to the server with the URL part "/classname/[basepart/]methodname" (without prefix). The methods "String
161  * web_uribase"
162  * returns the basepart of the URI
163  */
164  public final void registerObject( final Object p_object )
165  {
166  // URI begin
167  final String l_uriclass = c_seperator + p_object.getClass().getSimpleName().toLowerCase() + c_seperator;
168 
169  // get methods
170  for ( final Map.Entry<String, CReflection.CMethod> l_method : CReflection.getClassMethods(
171  p_object.getClass(), new CReflection.IMethodFilter()
172  {
173  @Override
174  public boolean filter( final java.lang.reflect.Method p_method )
175  {
176  return ( p_method.getName().toLowerCase().startsWith(
177  c_webstatic
178  ) || p_method.getName().toLowerCase().startsWith(
179  c_webdynamic
180  ) ) && ( !Modifier.isStatic( p_method.getModifiers() ) );
181  }
182  }
183  ).entrySet() )
184  {
185  this.bindStaticMethod( p_object, l_method.getValue(), l_uriclass );

268  @Override
269  public final Response serve( final IHTTPSession p_session )
270  {
271  // try to get the websocket first - try-catch avoid NPE because openWebSocket can create the exception
272  try
273  {
274  final Response l_response = m_websockethandler.serve( p_session );
275  if ( l_response != null )
276  return l_response;
277  }
278  catch ( final NullPointerException l_exception )
279  {
280  }
281 
282 
283  // no websocket
284  try
285  {
286  // get location - and check if it is a websocket
287  final IVirtualLocation l_location = m_virtuallocation.get( p_session );
288 
289  if ( l_location instanceof CVirtualDynamicMethod )
290  throw new Exception();
291  if ( l_location instanceof CVirtualStaticMethod )
292  return this.setDefaultHeader( this.getVirtualStaticMethod( l_location, p_session ), p_session );
293 
294  return this.setDefaultHeader( this.getVirtualDirFile( l_location, p_session ), p_session );
295 
296  }
297  catch ( final Throwable l_throwable )
298  {
299  CLogger.error( l_throwable );
300  return this.setDefaultHeader( new Response( Response.Status.INTERNAL_ERROR, MIME_PLAINTEXT, "ERROR 500\n" + l_throwable ), p_session );
301  }
302  }

361  /**
362  * reads the mime-type of an URL - first try to detect a mime-type which has no "application" prefix
363  */
364  @SuppressWarnings( "unchecked" )
365  private String getMimeType( final URL p_url )
366  {
367  final Collection l_types = m_mimetype.getMimeTypes( p_url );
368  if ( l_types.size() == 1 )
369  return l_types.iterator().next().toString();
370 
371  for ( final Object l_item : l_types )
372  {
373  final MimeType l_type = (MimeType) l_item;
374  if ( !l_type.toString().startsWith( "application/" ) )
375  return l_type.toString();
376  }
377  return MimeUtil2.UNKNOWN_MIME_TYPE.toString();
378  }

380  /**
381  * returns URI base if exists
382  *
383  * @param p_object object
384  * @return empty string or URI
385  */
386  private String getURIBase( final Object p_object )
387  {
388  // try to read basepart
389  String l_uribase = "";
390  try
391  {
392  final Object l_data = CReflection.getClassMethod( p_object.getClass(), c_weburibase ).getMethod().invoke( p_object );
393  if ( l_data instanceof String )
394  l_uribase = toURLFormat( l_data.toString() );
395 
396  if ( ( l_uribase != null ) && ( !l_uribase.isEmpty() ) )
397  {
398  if ( l_uribase.startsWith( c_seperator ) )
399  l_uribase = l_uribase.substring( 1 );
400  if ( !l_uribase.endsWith( c_seperator ) )
401  l_uribase += c_seperator;
402  }
403  }
404  catch ( final IllegalArgumentException | IllegalAccessException | InvocationTargetException l_exception )
405  {
406  }
407 
408  return l_uribase;
409  }

411  /**
412  * generates HTTP response of file & directory calls
413  *
414  * @param p_location location object
415  * @param p_session session object
416  * @return response
417  *
418  * @throws Throwable on error
419  */
420  private Response getVirtualDirFile( final IVirtualLocation p_location, final IHTTPSession p_session ) throws Throwable
421  {
422  final Response l_response;
423 
424  final URL l_physicalfile = p_location.get( p_session );
425  final String l_mimetype = this.getMimeType( l_physicalfile );
426  CLogger.info( p_session.getUri() + " " + l_physicalfile + " " + l_mimetype );
427 
428  switch ( FilenameUtils.getExtension( l_physicalfile.toString() ) )
429  {
430  case "htm":
431  case "html":
432  l_response = new Response(
433  Response.Status.OK, l_mimetype + "; charset=" + ( new InputStreamReader(
434  l_physicalfile.openStream()
435  ).getEncoding() ), l_physicalfile.openStream()
436  );
437  break;
438 
439  case "md":
440  if ( p_location.getMarkDownRenderer() != null )
441  l_response = new Response(
442  Response.Status.OK, p_location.getMarkDownRenderer().getMimeType(), p_location.getMarkDownRenderer().getHTML(
443  m_markdown, l_physicalfile
444  )
445  );
446  else
447  l_response = new Response( Response.Status.OK, l_mimetype, l_physicalfile.openStream() );
448  break;
449 
450  default:
451  l_response = new Response( Response.Status.OK, l_mimetype, l_physicalfile.openStream() );
452  }
453 
454  return l_response;
455  }
