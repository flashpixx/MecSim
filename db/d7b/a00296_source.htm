30 /**
31  * converting driving units
32  *
33  * @note [speed in meter/sec] * [timesampling in second] = [moved length in m] / ]cellsampling in meter] = moved cells ( km/h = 1000 / 3600 m/s )
34  */
35 public final class CUnitConvert
36 {
37 
38  /**
39  * constant value to scale km/h in m/s
40  **/
41  private static final double c_kmhInMs = 1000.0 / 3600.0;
42  /**
43  * constant value to scale m/s in km/h
44  **/
45  private static final double c_MsInKmh = 1 / c_kmhInMs;
46  /**
47  * cell size in meter
48  */
49  private final int m_cellsize = CConfiguration.getInstance().get().<Integer>get( "simulation/traffic/cellsampling" );
50  /**
51  * timestep in seconds
52  */
53  private final int m_timestep = CConfiguration.getInstance().get().<Integer>get( "simulation/traffic/timesampling" );
54  /**
55  * meter which can be moved at one timestep
56  */
57  private final double m_distancetimestep = c_kmhInMs * m_timestep;
58  /**
59  * number of cells which can be moved at one timestep
60  */
61  private final int m_celltimestep = (int) Math.floor( m_distancetimestep / m_cellsize );
62 
63 
64  /**
65  * returns the speed change in one timestep
66  *
67  * @param p_acceleration acceleration or deceleration in m/sec^2
68  * @return speed change in km/h
69  */
70  public final double getAccelerationToSpeed( final double p_acceleration )
71  {
72  return ( p_acceleration * c_MsInKmh * m_timestep );
73  }
74 
75  /**
76  * returns the cell size
77  *
78  * @return cell size in meter
79  */
80  public final int getCellSize()
81  {
82  return m_cellsize;
83  }
84 
85  /**
86  * returns cell number to distance in meter
87  *
88  * @param p_cells cell number
89  * @return meter value
90  */
91  public final double getCellToMeter( final int p_cells )
92  {
93  return p_cells * m_cellsize;
94  }
95 
96  /**
97  * returns cell number to distance in kilometer
98  *
99  * @param p_cells cell number
100  * @return kilometer value
101  */
102  public final double getCellToKiloMeter( final int p_cells )
103  {
104  return this.getCellToMeter( p_cells ) / 1000.0;
105  }
106 
107  /**
108  * returns the speed for traveling the given distance in one timestep
109  *
110  * @param p_distance distance in meter
111  * @return speed in km/h
112  */
113  public final int getSpeedOfDistance( final double p_distance )
114  {
115  return (int) ( p_distance / m_timestep * c_MsInKmh );
116  }
117 
118  /**
119  * returns the speed in cell positions at one timestep
120  *
121  * @param p_speed speed in km/h
122  * @return amount of cells / timestep
123  */
124  public final int getSpeedToCell( final int p_speed )
125  {
126  return p_speed * m_celltimestep;
127  }
128 
129  /**
130  * returns the distance in meter which
131  * can moved in one timestep
132  *
133  * @param p_speed speed in km/h
134  * @return meter
135  */
136  public final double getSpeedToDistance( final int p_speed )
137  {
138  return p_speed * c_kmhInMs * m_timestep;
139  }
140 
141  /**
142  * returns the timestep in seconds
143  *
144  * @return timestep

146  public final int getTime()
147  {

149  }
150 
151  /**

153  *
154  * @param p_step step number
155  * @return time in seconds
156  */
157  public final int getTime( final int p_step )
158  {
159  return m_timestep * p_step;
160  }
161 
162  /**

164  *
165  * @param p_step step number

168  public final double getTimeInMinutes( final int p_step )
169  {
170  return this.getTime( p_step ) / 60.0;
171  }
