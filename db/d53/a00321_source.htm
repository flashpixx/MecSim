114  /**
115  * creates the table structure
116  *
117  * @param p_tablename table name
118  * @param p_fields field list
119  * @param p_alter optional table alter commands
120  */
121  protected final void createTable( final String p_tablename, final String[] p_fields, final String[] p_alter )
122  {
123  final String l_tablename = this.getTableName( p_tablename );
124  try (
125  final Connection l_connect = c_datasource.getConnection()
126  )
127  {
128  final ResultSet l_result = l_connect.getMetaData().getTables( null, null, l_tablename, new String[]{"TABLE"} );
129  if ( !l_result.next() )
130  {
131  l_connect.createStatement().execute(
132  CLogger.info(
133  MessageFormat.format(
134  "create table {0} ({1})", l_tablename, String.join( ", ", p_fields )
135  )
136  )
137  );
138  if ( p_alter != null )
139  for ( final String l_item : p_alter )
140  l_connect.createStatement().execute( CLogger.info( MessageFormat.format( "alter table {0} {1}", l_tablename, l_item ) ) );
141  }
142  l_result.close();
143  }
144  catch ( final Exception l_exception )
145  {
146  m_active = false;
147  CLogger.error( l_exception );
148  }
149  }
